double R0 = 4;
double R1 = 1;
double AMP = 0.2;
double FREQ = 6;
double TURNS = 1;

double theta1 = u * 2 * Math.PI;
double phi1 = v * 2 * Math.PI;

double r   = R1 + AMP * Math.Sin(FREQ * phi1);
double ang = phi1 + pi*2 * Math.Sin(TURNS * theta1);

double tx = r * Math.Cos(ang) + R0;
x = tx * Math.Cos(theta1);
y = tx * Math.Sin(theta1);
z = r * Math.Sin(ang);
